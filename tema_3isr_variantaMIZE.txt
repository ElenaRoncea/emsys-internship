#define F_CPU 16000000UL
#define PORT_LED0 PORTD
#define PORT_LED1 PORTD
#define PORT_BUTTON PORTB
#define PIN_LED0 PIND4
#define PIN_LED1 PIND5
#define PIN_BUTTON PINB3




#include <avr/io.h>
#include <util/delay.h>

int main(void)
{
	setup();
	while (1) {
	PORTD ^= (1<<PIND4);
		if ((PINB & (1 << PINB3))==0){
			PORTD |= (1 << PIND5);
			_delay_ms(1000);
			
		}
		else{
			PORTD &= ~(1 << PIND5);
			_delay_ms(1000);
		}
		
	}
}

void setup(void){
	
	DDRB &= ~(1 << PINB3);
	PORTB |= (1 << PINB3);
	
	DDRD |= (1 << PIND5);
	
	DDRD |= (1<<PIND4);
	
}
